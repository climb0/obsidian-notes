@SpringBootApplication  
public class ObserverDemoApplication implements CommandLineRunner {  
  
    @Autowired  
    List<Observer> observerList;  
  
    public static void main(String[] args) {  
        SpringApplication.run(ObserverDemoApplication.class, args);  
    }  
  
    /**  
     * Spring容器启动后，立即执行一次  
     */  
    @Override  
    public void run(String... args) throws Exception {  
        sendMsg("bizType1", "业务1需要发送的内容");  
        sendMsg("bizType2", "业务2需要发送的内容");  
    }  
  
    private void sendMsg(String bizType, String content) {  
        observerList.forEach(observer -> observer.notify(bizType, content));  
    }  
  
}
